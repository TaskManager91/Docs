In order to use this functionality, every integrator should execute only 2 steps:
1. Add payengine.widget.min.js (get it here)
2. Call initAsModal() function on PayEngineWidget object to display the payment form in modal view
Here is snippet how your html could look like:
<!DOCTYPE html>
<html>
 <head>
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <script src="https://pp.payengine.de/widgetjs/payengine.widget.min.js"></script>
  <script>
   var callback = function (error, result) {
     if (error) {
       //There was an error
       //You will find more details in the error object
       return;
     }
     //Handle transaction successfully created
   }

   function openModal() {
     var publishableKey = "merchant_kdanvgjnhx";
     var orderId = "order_564c4d97f7";
     var optionalParameters = new Object;
     optionalParameters.products= ["creditcard", "sepa", "paypal", "paydirekt", "ratepay-directdebit", "ratepay-invoice"];
     optionalParameters.language = "en";
     optionalParameters.theme = "light";
     PayEngineWidget.initAsModal(publishableKey, orderId, optionalParameters, callback);
   }
  </script>
 </head>
 <body>
   //Your website html code here
   <button onclick="openModal()">Open modal</button>
 </body>
</html>
! Important -> Check out all mandatory and optional parameters in Modal Widget Reference.
